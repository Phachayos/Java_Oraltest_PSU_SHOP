⬡ OOP Concepts สรุปย่อ
01
Encapsulation
Field ทุกตัวประกาศเป็น private — ห้ามเข้าถึงจากภายนอกโดยตรง เข้าถึงได้ผ่าน getter/setter เท่านั้น totalSpent ไม่มี setter เลยเพื่อป้องกันการแก้ไข
02
Inheritance
Uniform และ Stationery ใช้ extends Product → รับ field และเมธอดมาทั้งหมด เรียก super() ใน constructor เพื่อให้คลาสแม่จัดการค่าก่อน
03
Polymorphism
ประกาศตัวแปรเป็น Product แต่ใส่ Object ชนิดลูกได้ เมื่อเรียก displayProductInfo() Java เลือก version ที่ถูกต้องเองตอน Runtime (Dynamic Dispatch)
04
Abstract Class
Product เป็น abstract class → new Product() ตรงๆ ไม่ได้ มีเมธอด abstract displayProductInfo() ที่บังคับให้คลาสลูกทุกตัว Override

💡

04 Abstract Class — คลาสนามธรรม
💡
หลักการ: Abstract Class คือ "แม่แบบ" ที่ไม่สามารถสร้าง Object ตรงๆ ได้ ใช้กำหนด structure ให้คลาสลูกทุกตัวต้องปฏิบัติตาม
ทำได้ ✓
Product p = new Uniform(...);
Product p = new Stationery(...);
extends Product แล้ว Override
ทำไม่ได้ ✗
new Product(...) → Compile Error!
extends แล้วไม่ Override → Error!

Product.java abstract class 
// ============================================================
// Product.java — Base Class (คลาสแม่)
// บทบาท: Abstract Class กลางที่สินค้าทุกชนิดต้องสืบทอด
// OOP: Encapsulation, Abstract Class
// ============================================================

public abstract class Product {   // abstract = สร้าง new Product() ตรงๆ ไม่ได้

    // ── Encapsulation: ประกาศ private ซ่อนข้อมูลจากภายนอก ──────
    private String id;       // รหัสสินค้า
    private String name;     // ชื่อสินค้า
    private double price;    // ราคาสินค้า

    // Constructor: คลาสลูกจะเรียกผ่าน super(id, name, price)
    public Product(String id, String name, double price) {
        this.id    = id;
        this.name  = name;
        this.price = price;
    }

    // ── Getter/Setter: ช่องทางเข้าถึง field private ────────────
    // Getter = อ่านค่า | Setter = เขียนค่า (ควบคุมได้)
    public String getId()              { return id; }
    public void setId(String id)       { this.id = id; }

    public String getName()            { return name; }
    public void setName(String name)   { this.name = name; }

    public double getPrice()           { return price; }
    public void setPrice(double price) { this.price = price; }

    // ── Abstract Method: บังคับให้คลาสลูกทุกตัว Override ────────
    // ไม่มี body {} = คลาสลูกต้องเขียนเองทั้งหมด
    // นี่คือหัวใจของ Polymorphism
    public abstract void displayProductInfo();   // ← abstract method
}
Uniform.java extends Product 
// ============================================================
// Uniform.java — Subclass ของ Product (ชุดนักศึกษา)
// OOP: Inheritance, Encapsulation, Polymorphism (Override)
// ============================================================

// "extends Product" = Inheritance: Uniform IS-A Product
public class Uniform extends Product {

    // field เพิ่มเติมเฉพาะของ Uniform (ที่ Product ไม่มี)
    private String size;   // Encapsulation: private

    // Constructor:
    // 1. เรียก super() ให้ Product จัดการ id, name, price ก่อน
    // 2. จากนั้นค่อยกำหนด size ของ Uniform เอง
    public Uniform(String id, String name, double price, String size) {
        super(id, name, price);   // Inheritance: ส่งให้ constructor ของ Product
        this.size = size;
    }

    // Getter/Setter สำหรับ size
    public String getSize()             { return size; }
    public void setSize(String size)   { this.size = size; }

    // ── Polymorphism: Override เมธอด abstract จาก Product ──────
    // @Override แสดงว่านี่คือการ Override (compiler จะตรวจให้)
    // เนื้อหาแสดงผลในรูปแบบของ Uniform โดยเฉพาะ
    @Override
    public void displayProductInfo() {   // Polymorphism
        System.out.println(
            "[Uniform] " + getName() +
            " (Size: " + size + ") | Price: " + getPrice() + " THB"
        );
    }
}
Stationery.java extends Product 
// ============================================================
// Stationery.java — Subclass ของ Product (เครื่องเขียน)
// OOP: Inheritance, Encapsulation, Polymorphism (Override)
// ============================================================

public class Stationery extends Product {   // Inheritance

    // field เพิ่มเติมเฉพาะของ Stationery (ประเภทเครื่องเขียน)
    private String type;   // Encapsulation: private

    // Constructor: ส่ง id, name, price ให้ Product ผ่าน super()
    public Stationery(String id, String name, double price, String type) {
        super(id, name, price);   // Inheritance: เรียก constructor แม่
        this.type = type;
    }

    // Getter/Setter สำหรับ type
    public String getType()             { return type; }
    public void setType(String type)   { this.type = type; }

    // ── Polymorphism: Override เมธอด abstract จาก Product ──────
    // เหมือน Uniform แต่รูปแบบข้อความต่างกัน
    // → ตัวอย่างชัดเจนว่า "เมธอดเดียวกัน ทำงานต่างกัน"
    @Override
    public void displayProductInfo() {   // Polymorphism
        System.out.println(
            "[Stationery] " + getName() +
            " (Type: " + type + ") | Price: " + getPrice() + " THB"
        );
    }
}
StudentCustomer.java class 
// ============================================================
// StudentCustomer.java — คลาสลูกค้านักศึกษา
// OOP: Encapsulation (totalSpent), Polymorphism (buyProduct)
// ============================================================

public class StudentCustomer {

    // ── Encapsulation: ทุก field เป็น private ──────────────────
    private String studentId;
    private String name;
    private double totalSpent;   // สำคัญ! ไม่มี setter → ภายนอกแก้ไม่ได้

    // Constructor: กำหนดค่าเริ่มต้น totalSpent = 0.0 ทันที
    public StudentCustomer(String studentId, String name) {
        this.studentId  = studentId;
        this.name       = name;
        this.totalSpent = 0.0;   // เริ่มที่ 0 เสมอ
    }

    // Getter เท่านั้น (studentId, name ไม่ควรเปลี่ยน)
    public String getStudentId()  { return studentId; }
    public String getName()        { return name; }
    public double getTotalSpent()  { return totalSpent; }   // อ่านได้อย่างเดียว

    // ── buyProduct: รับ Product (ประเภทแม่) ─────────────────────
    // → ใส่ Uniform หรือ Stationery ก็ได้ (Polymorphism)
    // → Java เลือก displayProductInfo() ที่ถูกต้องเองตอน Runtime
    public void buyProduct(Product product) {   // parameter เป็นชนิด Product (แม่)
        System.out.println(
            ">>> Student " + this.name +
            " (" + this.studentId + ") is buying:"
        );

        // Polymorphism: Java เลือกเรียก version ที่ถูกต้องเองตอน Runtime
        product.displayProductInfo();

        // สะสมยอดรวม — แก้ได้แค่จากภายในคลาสนี้ (Encapsulation)
        this.totalSpent += product.getPrice();

        System.out.println("-------------------------------------------------");
    }

    // แสดงสรุปใบเสร็จ
    public void displayTotalSummary() {
        System.out.println("=== Receipt Summary ===");
        System.out.println("Customer: "         + this.name);
        System.out.println("Total Amount Due: " + this.totalSpent + " THB");
        System.out.println("=======================");
    }
}
TestMain.java main class 
// ============================================================
// TestMain.java — Entry Point (จุดเริ่มต้นโปรแกรม)
// บทบาท: ทดสอบทุก OOP Concept ด้วย 5 Test Cases
// ============================================================

public class TestMain {
    public static void main(String[] args) {
        System.out.println("========== PSU Shop System ==========\n");

        // ── สร้างสินค้า ────────────────────────────────────────────
        // ตัวแปรประกาศเป็น Product (แม่) แต่ Object จริงเป็นลูก
        // → "Uniform IS-A Product" ทำให้ทำแบบนี้ได้ (Polymorphism + Inheritance)
        Product shirt = new Uniform("U001", "CoC Polo Shirt", 350.0, "XL");
        Product pen   = new Stationery("S001", "PSU Blue Pen", 15.25, "Ballpoint");
        Product paper = new Stationery("S002", "PSU Paper", 45.0, "A4");

        // สร้างนักศึกษา (totalSpent เริ่มที่ 0 อัตโนมัติ)
        StudentCustomer student = new StudentCustomer("6830614123", "SomChai");

        // ── Test Case 1: แสดงสินค้าทั้งหมด ─────────────────────────
        // เรียก displayProductInfo() บน Product reference
        // Java รู้เองว่าต้องเรียก version ใด (Polymorphism)
        System.out.println("--- Test Case 1: Product Catalog ---");
        shirt.displayProductInfo();   // → เรียก Uniform version
        pen.displayProductInfo();     // → เรียก Stationery version
        paper.displayProductInfo();   // → เรียก Stationery version

        // ── Test Case 2–4: ซื้อสินค้า ────────────────────────────────
        // buyProduct() รับ Product → ใส่ลูกได้ทุกชนิด (Polymorphism)
        // totalSpent สะสมอัตโนมัติ (Encapsulation)
        System.out.println("\n--- Test Case 2: Buy Uniform ---");
        student.buyProduct(shirt);

        System.out.println("--- Test Case 3: Buy Pen ---");
        student.buyProduct(pen);

        System.out.println("--- Test Case 4: Buy Notebook ---");
        student.buyProduct(paper);

        // ── Test Case 5: สรุปยอด ────────────────────────────────────
        // ยอด 350 + 15.25 + 45 = 410.25 THB
        // เข้าถึง totalSpent ได้แค่ผ่าน getTotalSpent() (Encapsulation)
        System.out.println("--- Test Case 5: Total Summary ---");
        student.displayTotalSummary();
    }
}
